# HELPERS

def ensure_dirs():
    os.makedirs(INBOUND_DIR, exist_ok=True)

def month_start_end(year, month):
    start = date(year, month, 1)
    if month == 12:
        end = date(year + 1, 1, 1) - timedelta(days=1)
    else:
        end = date(year, month + 1, 1) - timedelta(days=1)
    return start, end

def iter_months(start_year, start_month, count):
    y, m = start_year, start_month
    for _ in range(count):
        yield y, m
        m += 1
        if m == 13:
            m = 1
            y += 1

def random_date_in_month(year, month):
    start, end = month_start_end(year, month)
    delta_days = (end - start).days
    return start + timedelta(days=random.randint(0, delta_days))

def safe_write_csv(path, fieldnames, rows):
    # newline="" avoids blank lines on Windows
    with open(path, "w", newline="", encoding=CSV_ENCODING) as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        for r in rows:
            writer.writerow(r)

def pick_price(category):
    lo, hi = PRICE_RANGE_BY_CATEGORY[category]
    return round(random.uniform(lo, hi), 2)

# GENERATE STORES (2-3x more stores)
# ============================================================
def build_stores():
    stores = []
    store_count = 18  # adjust up/down; this is ~2-3x a tiny starter set
    store_names = [
        "Sunset Mall", "Pine Plaza", "Ridge Center", "Harbor Point",
        "Canyon Outlet", "Maple Square", "Rivergate", "Stonebrook",
        "Lakeside Market", "Grand Avenue", "Skyline Shops", "Copper Corner",
        "Meadow Walk", "North Gate", "South Point", "Westfield Hub",
        "Eastview Center", "Summit Square"
    ]

    # Build stores with Region->State->City
    sid = 1
    for name in store_names[:store_count]:
        region = random.choice(list(REGIONS_TO_STATES.keys()))
        state = random.choice(REGIONS_TO_STATES[region])
        city = random.choice(CITY_BY_STATE[state])
        stores.append({
            "StoreId": f"S{sid:04d}",
            "StoreName": name,
            "City": city,
            "State": state,
            "Region": region
        })
        sid += 1
    return stores

# ============================================================
# GENERATE PRODUCTS (more products per category + multiple brands)
# Pricing is consistent per ProductId
# ============================================================
def build_products():
    products = []
    pid = 1

    for category, names in PRODUCTS_BY_CATEGORY.items():
        brands = CATEGORY_BRANDS[category]
        for n in names:
            brand = random.choice(brands)
            product_id = f"P{pid:04d}"
            # Assign a stable base price per product so it does NOT change day-to-day
            base_price = pick_price(category)

            products.append({
                "ProductId": product_id,
                "ProductName": n,
                "Category": category,
                "Brand": brand,
                "BasePrice": base_price  # internal only; not written to products_master
            })
            pid += 1
    return products

# ============================================================
# SALES FILES (match your stg.SalesRaw columns)
# stg.SalesRaw has:
# SaleDate, ProductId, ProductName, Category, Brand, StoreId, StoreName,
# City, State, Units, Revenue, SourceFile, LoadDttm (latter two filled in SSIS)
# We'll output ONLY the CSV columns that the flat file expects.
# ============================================================
def build_sales_rows(products, stores, year, month):
    rows = []

    # Heavier volume: tune these for your laptop speed
    # Each month roughly: stores * products * avg 1-3 rows
    rows_per_store = 120  # adjust up/down

    # Quick lookup dicts
    prod_by_id = {p["ProductId"]: p for p in products}

    for s in stores:
        for _ in range(rows_per_store):
            p = random.choice(products)
            category = p["Category"]
            units_lo, units_hi = UNITS_RANGE_BY_CATEGORY[category]
            units = random.randint(units_lo, units_hi)

            # consistent pricing per product + small promo variance (optional)
            # promo variance kept small so price doesn't jump wildly
            base_price = p["BasePrice"]
            promo_factor = random.choice([1.00, 1.00, 1.00, 0.95, 1.05])  # mostly stable
            unit_price = round(base_price * promo_factor, 2)

            revenue = round(units * unit_price, 2)

            sale_dt = random_date_in_month(year, month)

            rows.append({
                "SaleDate": sale_dt.strftime("%Y-%m-%d"),
                "ProductId": p["ProductId"],
                "ProductName": p["ProductName"],
                "Category": p["Category"],
                "Brand": p["Brand"],
                "StoreId": s["StoreId"],
                "StoreName": s["StoreName"],
                "City": s["City"],
                "State": s["State"],
                "Units": str(units),
                # Important: plain numeric with dot decimal, no commas, no currency symbol
                "Revenue": f"{revenue:.2f}"
            })

    return rows
